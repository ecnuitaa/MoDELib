145  SegmentMeshCollisionPairContainerType planeMeshIntersection() const
146  {
150  SegmentMeshCollisionPairContainerType temp;
151  if (shared.use_boundary)
152  {
153  const VectorDimD x0=planeNormal*height;
154  VectorDimD n=planeNormal;
155 
156  for(typename SimplexObserver<dim,dim-1>::const_iterator fIter =SimplexObserver<dim,dim-1>::simplexBegin();
157  /* */ fIter!=SimplexObserver<dim,dim-1>::simplexEnd();++fIter)
158  {
159  if(fIter->second->isBoundarySimplex())
160  {
161  std::deque<VectorDimD> intersectionPoints;
162  for(unsigned int e=0;e<dim;++e)
163  {// this part of the code is specific to dim=3;
164  const VectorDimD& v0(fIter->second->child(e).child(0).P0);
165  const VectorDimD& v1(fIter->second->child(e).child(1).P0);
166  const double u ((x0-v0).dot(n) / (v1-v0).dot(n));
167  if(u>=0 && u<=1.0)
168  {
169  const VectorDimD P(v0 + u*(v1-v0));
170  bool isDifferent=true;
171  for (unsigned int k=0;k<intersectionPoints.size();++k)
172  {
173  isDifferent*= (P-intersectionPoints[k]).squaredNorm()>FLT_EPSILON;
174  }
175 
176  if (isDifferent)
177  {
178  intersectionPoints.emplace_back(P);
179  }
180  }
181  }
182  assert(intersectionPoints.size()<3);
183  if (intersectionPoints.size()==2)
184  {
185  temp.emplace_back(intersectionPoints[0],intersectionPoints[1]);
186  }
187  }
188  }
189  }
190  return temp;
191  }
