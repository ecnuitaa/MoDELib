301  template<typename Condition,typename DislocationNetworkType>
302  void addDirichletCondition(const size_t& nodeListID,
303  const Condition& cond,
304  const std::array<bool,dofPerNode>& constrainDof,
305  const DislocationNetworkType& DN)
306  {
309  const auto t0= std::chrono::system_clock::now();
310  model::cout<<"adding DirichletCondition... "<<std::flush;
311  // Add the Dirichlet condition
312  u->addDirichletCondition(nodeListID,cond,constrainDof);
313 
314  if(apply_DD_displacement)
315  {
316  // Compute the DislocationNetwork displacement at nodeList
317  model::cout<<"subtracting DislocationDisplacement... "<<std::flush;
318  typedef BoundaryDisplacementPoint<DislocationNetworkType> FieldPointType;
319  typedef typename FieldPointType::DisplacementField DisplacementField;
320  std::deque<FieldPointType> fieldPoints; // the container of field points
321 
322  for (auto node : fe->nodeList(nodeListID)) // range-based for loop (C++11)
323  {
324  // Compute S vector
325  Eigen::Matrix<double,dim,1> s(Eigen::Matrix<double,dim,1>::Zero());
326  for(auto ele : *node)
327  {
328  const Eigen::Matrix<double,dim+1,1> bary(ele->simplex.pos2bary(node->P0));
329  for(int k=0;k<dim+1;++k)
330  {
331  if (std::fabs(bary(k))<FLT_EPSILON && ele->simplex.child(k).isBoundarySimplex())
332  {
333  s += ele->simplex.nda.col(k).normalized();
334  }
335  }
336  }
337 
338  const double sNorm(s.norm());
339  assert(sNorm>0.0 && "s-vector has zero norm.");
340  fieldPoints.emplace_back(*node,s/sNorm);
341  }
342  DN.template computeField<FieldPointType,DisplacementField>(fieldPoints);
343 
344  // Subtract the DislocationNetwork displacement from the Dirichlet conditions
345  for(size_t n=0;n<fieldPoints.size();++n)
346  {
347  for(int dof=0;dof<dofPerNode;++dof)
348  {
349  if(constrainDof[dof])
350  {
351  u->dirichletConditions().at(fe->nodeList(nodeListID)[n]->gID*dofPerNode+dof) -= fieldPoints[n].template field<DisplacementField>()(dof);
352  }
353  }
354  }
355 
356  if(DN.shared.use_virtualSegments) // Add solid angle contribution
357  {
358  //std::cout<<"NEED TO COMPUTE DISPLACEMENT OF RADIAL SEGMENTS"<<std::endl;
359 
360  for(size_t n=0;n<fieldPoints.size();++n)
361  {
362  VectorDim dispJump(VectorDim::Zero());
363 
364  for (typename DislocationNetworkType::NetworkLinkContainerType::const_iterator linkIter=DN.linkBegin();linkIter!=DN.linkEnd();++linkIter)
365  {
366  linkIter->second.addToSolidAngleJump(fieldPoints[n].P,fieldPoints[n].S,dispJump);
367  }
368 
369  for(int dof=0;dof<dofPerNode;++dof)
370  {
371  if(constrainDof[dof])
372  {
373  u->dirichletConditions().at(fe->nodeList(nodeListID)[n]->gID*dofPerNode+dof) -= dispJump(dof);
374  }
375  }
376  }
377  } // end virtual loops
378  }
379 
380  model::cout<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t0)).count()<<" sec]"<<defaultColor<<std::endl;
381  }
