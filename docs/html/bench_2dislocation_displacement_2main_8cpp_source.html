37 {
42  Eigen::Matrix<double,3,1> disp(Eigen::Matrix<double,3,1>::Zero()); // 3x3 matrix of zeros
43 
44  const double& x(P(0));
45  const double& y(P(1));
46  const double x2(x*x);
47  const double y2(y*y);
48  const double R2(x2+y2);
49  const double& nu(Material<Isotropic>::nu);
50 
51  double at2yx=atan2(y,x); // atan2 output results in [-pi;pi]
52  if(at2yx<0.0)
53  {// transform at2yx in [0;2*pi]
54  at2yx+=2.0*M_PI;
55  }
56  at2yx-=M_PI; // divide equally jump across slip plane
57 
58  // From Hirth and Lothe, Therory of Dislocations 2nd Ed, p 78. Eq. (3-45) and (3.56)
59  // disp(0)= at2yx + x*y/(2.0*(1.0-nu)*R2);
60  // disp(1)= -((1.0-2.0*nu)/(4.0*(1.0-nu))*log(R2)+(x2-y2)/(4.0*(1.0-nu)*R2));
61 
62  // Eshelby (Brit J Appl Phys 1966 v.17)
63  disp(0)= at2yx - x*y/(2.0*(1.0-nu)*R2);
64  disp(1)= -(1.0-2.0*nu)/(4.0*(1.0-nu))*log(R2)+y2/(2.0*(1.0-nu)*R2);
65  // disp(1)= -(1.0-2.0*nu)/(4.0*(1.0-nu))*log(x2+y2)+(x2-y2)/(4.0*(1.0-nu)*R2);
66  disp *= 1.0/(2.0*M_PI);
67 
68  //From Nabarro, p.57ttttt
69  return disp;
70 }
