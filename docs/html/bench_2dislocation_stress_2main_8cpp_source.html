22  Eigen::Matrix<double,3,3> stress(Eigen::Matrix<double,3,3>::Zero()); // 3x3 matrix of zeros
23 
24  // From Hirth and Lothe ch 3.4, p76.
25  // Assume unitary b and mu
26 
27  const double& x(P(0));
28  const double& y(P(1));
29  const double x2(x*x);
30  const double y2(y*y);
31  const double R4(std::pow(x2+y2,2));
32 
33  stress(0,0)=-y*(3.0*x2+y2)/R4; // sigma_xx
34  stress(1,1)= y*( x2-y2)/R4; // sigma_yy
35  stress(2,2)=-2.0*Material<Isotropic>::nu*y/(x2+y2); // sigma_zz
36  stress(0,1)= x*( x2-y2)/R4; // sigma_xy
37  stress(1,0)=stress(0,1); // symmetry
38
