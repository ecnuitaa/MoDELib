231 
232  // The polynomial coefficients of this spine segment
233  Eigen::Matrix<double,dim,Ncoeff> coeffs(this->polynomialCoeff());
234  coeffs.col(0)-=P0;
235 
236 
237  // The derivative of the polynomial coefficients
238  Eigen::Matrix<double,dim,Ncoeff-1> dcoeffs(Eigen::Matrix<double,dim,Ncoeff-1>::Zero());
239  for (int i=0;i<Ncoeff-1;++i)
240  {
241  dcoeffs.col(i)=(i+1)*coeffs.col(i+1);
242  }
243 
244  Eigen::Matrix<double,1,2*Ncoeff-2> pcoeffs(Eigen::Matrix<double,1,2*Ncoeff-2>::Zero()); // degree of product = pOrder+(pOrder-1)=2*pOrder-1. nCoeffs of product = 2*pOrder-1+1= 2*pOrder = 2*Ncoeff-2
245 
246  // The polynomial coefficients of (P-P0)*dP/du, in reverse order
247  for (int i=0;i<Ncoeff;++i)
248  {
249  for (int j=0;j<Ncoeff-1;++j)
250  {
251  pcoeffs(2*Ncoeff-3-i-j) += coeffs.col(i).dot(dcoeffs.col(j));
252  }
253  }
