50  static Eigen::MatrixXd abcsissasAndWeights(const int& qOrder)
51  {
52  std::cout<<"computing qOrder="<<qOrder<<"..."<<std::flush;
53 
54  // Create and fill the Jacobi upper triangular matrix
55  Eigen::MatrixXd jUT = Eigen::MatrixXd::Zero(qOrder,qOrder);
56  for (int n=0;n<qOrder;++n)
57  {
58  int k = n+1;
59  jUT(n,n) = -PolynomialType::b(k)/PolynomialType::a(k);
60  if (n<qOrder-1)
61  {
62  jUT(n,n+1)= std::sqrt(PolynomialType::c(k+1)/PolynomialType::a(k)/PolynomialType::a(k+1));
63  }
64  }
65 
66  // Create a SelfAdjointEigenSolver from the self-adjoint part of jUT
67  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(jUT.template selfadjointView<Eigen::Upper>());
68  Eigen::VectorXd eigenVal = es.eigenvalues();
69  Eigen::MatrixXd eigenVec = es.eigenvectors();
70 
71  // Insert each abscissa-weight pair in std::set for automatic sorting with increasing abscissas
72  typedef std::set<std::pair<double,double> > EigenSetType;
73  typedef EigenSetType::const_iterator EigenSetIterType;
74  EigenSetType eigenSet;
75  for (int k=0;k<qOrder;++k)
76  {
77  eigenSet.insert( std::make_pair( (eigenVal(k)+1.0)*0.5, std::pow(eigenVec(0,k),2) ) );
78  }
79 
80  // Copy the sorted values to abscissas_ and weigts_
81  Eigen::MatrixXd abcsissasAndWeights_(Eigen::MatrixXd::Zero(dim+1,qOrder));
82  int k=0;
83  for (EigenSetIterType iter=eigenSet.begin(); iter!=eigenSet.end();++iter)
84  {
85  abcsissasAndWeights_(0,k)= iter->first;

88  }
