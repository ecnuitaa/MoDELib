97  scalarType g = std::pow(chord.norm(),alpha);
98 
99  for (int k=0;k<Np;++k){
100  scalarType u1=k*1.0/(Np-1);

102  scalarType u3=u2*u1;

104  // Compute positions along axis
105  tubeAxis.col(k) = ( 2.0f*u3-3.0f*u2+1.0f) * P0T0P1T1BN.col(0)

107  /*************/ + (-2.0f*u3+3.0f*u2) * P0T0P1T1BN.col(2)
