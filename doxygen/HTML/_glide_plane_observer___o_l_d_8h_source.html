84  const GlidePlaneType* const p2)
85  {/*@param[in] p1 first plane
86  *@param[in] p2 second plane
87  *\returns the infinite line of interseciton between the two planes,
88  * if already computed. Otherwise it computes the interseciton, stores it,
89  * and returns it.
90  */
91  const auto key=std::make_pair(std::max(p1->sID,p2->sID),std::min(p1->sID,p2->sID));
92  const auto iter=glidePlaneIntersections().find(key);
93  if(iter==glidePlaneIntersections().end())

95  const bool success=glidePlaneIntersections().emplace(std::piecewise_construct,
96  std::make_tuple(key),
97  std::make_tuple(p1->P,p1->unitNormal,p2->P,p2->unitNormal)
98  ).second;
99  assert(success);
100  }
101  return glidePlaneIntersections().at(key);
102  }
