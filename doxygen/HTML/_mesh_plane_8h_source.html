40  static Plane<dim> getPlaneBetweenRegions(const SimplicialMesh<dim>& mesh,
41  const int& rID1,
42  const int& rID2)
43  {
44 
45  const auto& regionBnd(mesh.regionBoundary(rID1,rID2));
46 
47  VectorDim N(VectorDim::Zero());
48  VectorDim P(VectorDim::Zero());
49  size_t k=0;
50  for(const auto& triangle : regionBnd)
51  {
52 
53  const Simplex<dim,dim>& tet(**triangle->parents().begin());
54  const size_t faceID=tet.childOrder(triangle->xID);
55  const VectorDim n=tet.nda.col(faceID);
56 
57  if(n.dot(N)>=0.0)
58  {
59  N+=n.normalized();
60  }
61  else
62  {
63  N-=n.normalized();
64  }
65 
66  const auto vertices= triangle->vertices();
67  for(const auto& vertex : vertices)
68  {
69  P+=vertex->P0;
70  k++;
71  }
72  }
73  Plane<dim> plane(P/k,N);
74 
75  // Check that all tringle vertices are contained by both GB planes
76  for(const auto& triangle : regionBnd)
77  {
78  const auto vertices= triangle->vertices();
79  for(const auto& vertex : vertices)
80  {
81  assert(plane.contains(vertex->P0) && "TRIANGLE VERTEX NOT CONTAINED IN GBPLANE");
82  }
83  }
84 
85  return plane;
86  }

89  static PlaneMeshIntersectionContainerType getBoundaryBetweenRegions(const SimplicialMesh<dim>& mesh,
90  const int& rID1,
91  const int& rID2)
92  {
93  assert(dim==3 && "ALGORITHM ONLY VALID IN dim=3");
94  // Normal to plane
95 
96  // Get unsorted boundary edges
97  std::set<const Simplex<dim,dim-2>*> ub=mesh.regionBoundary(rID1,rID2).unsortedBoundary();
98 
99  // Compute center of plane
100  VectorDim c(VectorDim::Zero());
101  for(const auto& edge : ub)
102  {
103  c+=edge->child(0).P0;
104  c+=edge->child(1).P0;
105  }
106  c/=(2*ub.size()); //center of the plane
107 
108  // Rotation matrix to local reference system
109  const VectorDim x3((*mesh.regionBoundary(rID1,rID2).begin())->outNormal(rID1).normalized());
110  const VectorDim x1(((*ub.begin())->child(0).P0-c).normalized());
111  Eigen::Matrix<double,dim,dim> R;
112  R.col(0)=x1;
113  R.col(1)=x3.cross(x1);
114  R.col(2)=x3;
115 
116  // Sort the edges
117  std::map<double,std::pair<const Simplex<dim,dim-2>* const,VectorDim>> sortedEdges;
118  for(const auto& edge : ub)
119  {
120  const VectorDim x0=R.transpose()*(edge->child(0).P0-c);
121  const double angle0=atan2(x0(1),x0(0));
122  sortedEdges.emplace(angle0,std::make_pair(edge,edge->child(0).P0));
123 
124  const VectorDim x1=R.transpose()*(edge->child(1).P0-c);
125  const double angle1=atan2(x1(1),x1(0));
126  sortedEdges.emplace(angle1,std::make_pair(edge,edge->child(1).P0));
127  }
128 
129  assert(sortedEdges.size()==ub.size());
130 
131  PlaneMeshIntersectionContainerType temp;
132  for(const auto& pair : sortedEdges)
133  {
134  temp.push_back(pair.second);
135  }
136 
137  return PlaneMeshIntersection<dim>::reducedPlaneMeshIntersection(temp);
138  }

141  static PlaneMeshIntersectionContainerType getPlaneIntersection(const SimplicialMesh<dim>& mesh,
142  const int& rID,
143  const VectorDim& P0,
144  const VectorDim& N)
145  {
146 
147  const double nNorm(N.norm());
148  assert(nNorm>FLT_EPSILON);
149  const VectorDim n(N/nNorm);
150 
151  RootContainerType rootDeq;
152  for(const auto& edge : mesh.template observer<1>())
153  {
154  if( edge.second->isBoundarySimplex()
155  || edge.second->isRegionBoundarySimplex())
156  {
157  if(edge.second->isInRegion(rID))
158  {
159  const VectorDim& v0(edge.second->child(0).P0);
160  const VectorDim& v1(edge.second->child(1).P0);
161 
162  // check intersection of v0->v1 with plane
163  // x=v0+u(v1-v0)
164  // (x-P0).n=0
165  // (v0+u(v1-v0)-P0).n=0
166  // u=(P0-v0).n/(v1-v0).n;
167  const double edgeNorm=(v1-v0).norm();
168  assert(edgeNorm>FLT_EPSILON && "mesh edge has zero norm.");
169  const double den=(v1-v0).dot(n);
170  const double num=(P0-v0).dot(n);
171  const double P0v0norm=(P0-v0).norm();
172 
173  const double numCheck= (P0v0norm<FLT_EPSILON)? 0.0 : num/P0v0norm;
174 
175  if (fabs(den/edgeNorm)>FLT_EPSILON)
176  {
177  // edge intersects plane
178  const double u=num/den;
179 
180  if(fabs(u)<FLT_EPSILON)
181  {
182  rootDeq.emplace_back(v0,edge.second);
183  }
184  else if (u>=FLT_EPSILON && u<=1.0-FLT_EPSILON)
185  {
186  rootDeq.emplace_back((1.0-u)*v0 + u*v1,edge.second);
187  }
188  else if (fabs(1.0-u)<FLT_EPSILON)
189  {
190  rootDeq.emplace_back(v1,edge.second);
191  }
192  else
193  {// no roots
194 
195  }
196 
197  }
198  else
199  {
200  if (fabs(numCheck)>FLT_EPSILON)
201  {// edge is parallel to plane, no intersection
202 
203  }
204  else
205  {// edge is coplanar
206  rootDeq.emplace_back(v0,edge.second);
207  rootDeq.emplace_back(v1,edge.second);
208  }
209  }
210  }
211  }
212  }
213 // std::cout<<"MeshPlane rootDeq: "<<rootDeq.size()<<std::endl;
214 
215  PlaneMeshIntersectionContainerType temp;
216 
217  if(rootDeq.size())
218  {
219 
220  // compute center
221  VectorDim c(VectorDim::Zero());
222  for(const auto& pair : rootDeq)
223  {
224  c+=pair.first;
225  }
226  c/=rootDeq.size(); //center of the plane
227 
228  // Check that points belong to a plane
229  for(const auto& pair : rootDeq)
230  {
231  assert(fabs(n.dot(c-pair.first))<FLT_EPSILON);
232  }
233 
234  const VectorDim refDir(rootDeq[0].first-c);
235  const double refDirNorm(refDir.norm());
236  assert(refDirNorm>FLT_EPSILON);
237 
238  // Local rotation matrix
239  Eigen::Matrix<double,dim,dim> R;
240  R.col(0)=refDir/refDirNorm;
241  R.col(2)=n;
242  R.col(1)=R.col(2).cross(R.col(0));
243 
244 
245  assert((R*R.transpose()-MatrixDim::Identity()).norm()<FLT_EPSILON);
246  assert(fabs(R.determinant()-1.0)<FLT_EPSILON);
247 
248  std::map<double,std::pair<const Simplex<dim,dim-2>* const,VectorDim>> sortedEdges;
249 
250  for(const auto& pair : rootDeq)
251  {
252  const VectorDim x0=R.transpose()*(pair.first-c);
253  const double angle0=atan2(x0(1),x0(0));
254  sortedEdges.emplace(angle0,std::make_pair(pair.second,pair.first));
255  }
256 
257  for(const auto& pair : sortedEdges)
258  {
259  temp.push_back(pair.second);
260  }
261 
262  }
263 
264 
265  return PlaneMeshIntersection<dim>::reducedPlaneMeshIntersection(temp);
266  }
