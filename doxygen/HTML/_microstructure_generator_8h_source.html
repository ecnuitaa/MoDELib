109  void addIrradiationLoops()
110  {
111 
112  std::mt19937 generator;
113 
114 
115  size_t ndefects=0;
116  double defectsDensity=ndefects/mesh.volume()/std::pow(Material<Isotropic>::b_real,3);
117  while(defectsDensity<targetIrradiationLoopDensity)
118  {
119 
120 
121  const std::pair<LatticeVector<dim>,int> rp=this->randomPointInMesh();
122  const int& grainID=rp.second; // random grain ID
123  const LatticeVector<dim>& L0=rp.first; // random lattice position in the grain
124  const VectorDimD P0(L0.cartesian()); // cartesian position of L0
125 
126  std::uniform_int_distribution<> distribution(0,poly.grain(grainID).slipSystems().size()-1);
127  const int rSS=distribution(generator); // a random SlipSystem ID
128  const SlipSystem& slipSystem=this->poly.grain(grainID).slipSystems()[rSS];
129  const VectorDimD b=slipSystem.s.cartesian(); // Burgers vector
130  const VectorDimD a(b.normalized());
131  //const VectorDimD c(slipSystem.s.cartesian());
132 
133  std::vector<VectorDimD> points;
134 
135  const int NP=6;
136  for(int k=0;k<NP;++k)
137  {
138  points.push_back(P0+Eigen::AngleAxis<double>(k*2.0*M_PI/NP,a)*slipSystem.unitNormal*0.5*averageLoopSize/Material<Isotropic>::b_real);
139  }
140 
141  bool pointsIncluded=true;
142  for(const auto& point : points)
143  {
144  pointsIncluded*=mesh.searchRegion(grainID,point).first;
145  }
146 
147  if(pointsIncluded)
148  {
149 
150  for(int k=0;k<NP;++k)
151  {
152  nodesIO.emplace_back(nodeID+k,points[k],Eigen::Matrix<double,1,3>::Zero(),1.0,snID,0);
153 
154  const int nextNodeID=(k+1)<NP? this->nodeID+k+1 : this->nodeID;
155  edgesIO.emplace_back(this->loopID,this->nodeID+k,nextNodeID,0);
156 
157  }
158 
159  loopsIO.emplace_back(this->loopID+0, b,a,P0,grainID);
160 
161 
162  snID++;
163  loopID++;
164  nodeID+=NP;
165  ndefects++;
166  defectsDensity=ndefects/mesh.volume()/std::pow(Material<Isotropic>::b_real,3);
167  std::cout<<"irradiation defects density="<<defectsDensity<<std::endl;
168  }
169  }

342  const VectorDimD& P1,
343  const VectorDimD& D,
344  const PlaneMeshIntersectionContainerType& pp)
345  {
346 
347 
348 
349  const double dNorm(D.norm());
350  assert(dNorm>FLT_EPSILON);
351  const VectorDimD dir=D/dNorm;
352  // Let a point v on the boundary be written as v=P0+u1*(P1-P0)+u2*d
353  // then we have [P1-P0 d]*[u1 u2]^T=v-P0
354 
355  Eigen::Matrix<double,3,2> A;
356  A.col(0)=P1-P0;
357  A.col(1)=dir;
358  const Eigen::LLT<Eigen::Matrix<double,2,2>> llt(A.transpose()*A);
359  assert(llt.info()==Eigen::Success);
360 
361  std::map<double,VectorDimD> temp; // keep points sorted by parameter u1
362  for(size_t m=0;m<pp.size();++m)
363  {
364  const Eigen::Matrix<double,2,1> x=llt.solve(A.transpose()*(pp[m].second-P0));

377  static std::pair<int,VectorDimD> boundaryProjection(const VectorDimD& P,
378  const VectorDimD& D,
379  const PlaneMeshIntersectionContainerType& pp)
380  {
381  const double dNorm(D.norm());
382  assert(dNorm>FLT_EPSILON);
383  const VectorDimD dir=D/dNorm;
384  // line1 is P+u1*dir, u>0
385 
386  bool success=false;
387  std::pair<int,VectorDimD> temp=std::make_pair(-1,VectorDimD::Zero());
388 
389  for(size_t k=0;k<pp.size();++k)
390  {
391  const size_t k1 = ((k==(pp.size()-1))? 0 : k+1);
392  const VectorDimD& v0=pp[k].second;
393  const VectorDimD& v1=pp[k1].second;
394  // line2 is v0+u2*(v1-v0), 0<=u2<=1
395 
396  //P+u1*dir=v0+u2*(v1-v0)
397  // [dir -(v1-v0)] [u1 u2] = [v0-P]
398  // In least square sense
