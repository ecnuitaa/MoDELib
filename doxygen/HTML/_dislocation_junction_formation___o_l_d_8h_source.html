59  void bringBackToMesh(VectorDimD& P,
60  const std::pair<bool,const Simplex<dim,dim>*>& search,
61  const VectorDimD& c,
62  const VectorDimD& g) const
63  {
64 
65  const VectorDimD dir(c.cross(g));
66  assert(dir.squaredNorm()>FLT_EPSILON && "bringBackToMesh: direction has zero norm.");
67  const VectorDimD d(dir.normalized());
68 
69  int faceID;
70  search.second->pos2bary(P).minCoeff(&faceID); // find the ID of the face with minimum bary coordinate
71  const bool isBoundaryFace(search.second->child(faceID).isBoundarySimplex());
72  assert(isBoundaryFace && "bringBackToMesh: face is not a boundary face.");
73 
74  const VectorDimD V(search.second->child(faceID).vertices()[0]->P0);
75  const VectorDimD N(search.second->nda.col(faceID).normalized());
76  const double den(d.dot(N));
77  if(std::fabs(den)<FLT_EPSILON)
78  {
79  model::cout<<"d="<<d.transpose()<<std::endl;
80  model::cout<<"N="<<N.transpose()<<std::endl;
81  assert(0 && "bringBackToMesh: direction is parallel to mesh face.");
82  }
83 
84  const double u((V-P).dot(N)/den);
85 
86  P += u*d;
87  }
