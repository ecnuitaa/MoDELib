53  const double& L,
54  const double& dt)
55  {
56 #ifdef _OPENMP
57  return distributions[omp_get_thread_num()](generators[omp_get_thread_num()])*sqrt(2.0*kB*T/B/L/dt);
58 #else
59  return distributions[0](generators[0])*sqrt(2.0*kB*T/B/L/dt);
60 #endif

205  static double sigmoid(const double & x)
206  {
207  return 1.0/(1.0+exp(-x));
208  }

210  /**********************************************************************/
211  double velocity(const MatrixDim& S,
212  const VectorDim& b,
213  const VectorDim& xi,
214  const VectorDim& n,
215  const double& T,
216  const double& dL,
217  const double& dt,
218  const bool& use_stochasticForce) const
219  {
220 
221 
222  const double bNorm=b.norm();
223  const VectorDim s = b/bNorm;
224  const VectorDim n1 = Eigen::AngleAxisd(M_PI/3.0,s)*n;
225 
226  // Compute components of non-Schmid model
227  const double tau=std::fabs(s.transpose()*S*n); // magnitude of resolved shear stress
228  const double tauOrt=n.cross(s).transpose()*S*n;
229  const double tau1=std::fabs(s.transpose()*S*n1); // resolved schear stress on
230  const double tauOrt1=n1.cross(s).transpose()*S*n1;
231 
232  const double num=tau+a1*tau1;
233  //
234  assert(num>=0.0 && "num must be >= 0.");
235  const double den=(1.0+0.5*a1)*tauC*(a4+a0*sigmoid(-(a2*tauOrt+a3*tauOrt1)/tauC));
236  assert(den>0.0 && "den must be > 0.");
237 
238  const double Theta=num/den;
239  const double dg = (Theta<1.0)? (std::pow(1.0-std::pow(Theta,p),q)-T/T0) : 0.0;
240  const double dg1 = (dg>0.0)? dg : 0.0;
241  const double expCoeff = exp(-dH0*dg1/(2.0*kB_eV*T));
242 
243  // Compute screw drag coeff
244  const double sgm=sigmoid((0.05-dg1)/0.05);
245  const double Bs=Bk*w/(2.0*h)*(1.0-sgm)+(B0s+B1s*T)*sgm;
246 
247  // Compute screw velocity
248  double vs=tau*bNorm/Bs*expCoeff;
249 
250  // Compute edge velocity
251  double ve=tau*bNorm/(B0e+B1e*T);
252 
253  if(use_stochasticForce)
254  {
255  vs+=StochasticForceGenerator::velocity(kB,T,Bs,dL,dt);
256 
257  ve+=StochasticForceGenerator::velocity(kB,T,B0e+B1e*T,dL,dt);
258  }
259 
260  // Interpolate ve and vs
261  const double cos2=std::pow(b.normalized().dot(xi),2);
262  const double sin2=1.0-cos2;
263  return vs*cos2+ve*sin2;
264  }
