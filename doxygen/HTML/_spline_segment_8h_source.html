283  std::pair<double,std::pair<double,VectorDim> > closestPoint(const VectorDim& P0) const
284  {
292  // solve (P-P0)*dP/du=0
293 
294  // The polynomial coefficients of this spine segment
295  Eigen::Matrix<double,dim,Ncoeff> coeffs(this->polynomialCoeff());
296  coeffs.col(0)-=P0;
297 
298 
299  // The derivative of the polynomial coefficients
300  Eigen::Matrix<double,dim,Ncoeff-1> dcoeffs(Eigen::Matrix<double,dim,Ncoeff-1>::Zero());
301  for (int i=0;i<Ncoeff-1;++i)
302  {
303  dcoeffs.col(i)=(i+1)*coeffs.col(i+1);
304  }
305 
306  Eigen::Matrix<double,1,2*Ncoeff-2> pcoeffs(Eigen::Matrix<double,1,2*Ncoeff-2>::Zero()); // degree of product = pOrder+(pOrder-1)=2*pOrder-1. nCoeffs of product = 2*pOrder-1+1= 2*pOrder = 2*Ncoeff-2
307 
308  // The polynomial coefficients of (P-P0)*dP/du, in reverse order
309  for (int i=0;i<Ncoeff;++i)
310  {
311  for (int j=0;j<Ncoeff-1;++j)
312  {
313  pcoeffs(2*Ncoeff-3-i-j) += coeffs.col(i).dot(dcoeffs.col(j));
314  }
315  }
316 
317  // Compute roots using the eigenvalue method
318  MatrixCompanion<2*Ncoeff-3> mc(pcoeffs);
319 
320  // sort roots according to distance to P0
321  std::map<double,std::pair<double,VectorDim>,std::less<double>, Eigen::aligned_allocator<std::pair<double,std::pair<double,VectorDim>> > > rootMap;
322 
323  // for (int k=0;k<2*Ncoeff-3;++k)
324  for (size_t k=0;k<mc.rootSize;++k)
325  {
326  if (std::fabs(mc.root(k).imag())<FLT_EPSILON && mc.root(k).real()>0.0 && mc.root(k).real()<1.0 )
327  {
328 
329  VectorDim P(this->get_r(mc.root(k).real()));
330  rootMap.insert(std::make_pair((P-P0).norm(), std::make_pair(mc.root(k).real(),P) ));
331 
332  }
333 
334  }
335 
336  // check distance to limits of interval
337  rootMap.insert(std::make_pair((this->source->get_P()-P0).norm(), std::make_pair(0.0,this->source->get_P()) ));
338  rootMap.insert(std::make_pair((this-> sink->get_P()-P0).norm(), std::make_pair(1.0,this-> sink->get_P()) ));
339 
340  return *rootMap.begin();
341 
342  }
