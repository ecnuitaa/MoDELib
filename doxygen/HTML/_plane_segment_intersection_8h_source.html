32  static SolutionType findIntersection(const VectorDimD& P0,
33  const VectorDimD& Normal,
34  const VectorDimD& v0,
35  const VectorDimD& v1)
36  {
37  const double nNorm(Normal.norm());
38  assert(nNorm>FLT_EPSILON);
39  const VectorDimD n(Normal/nNorm);
40 
41  // check intersection of v0->v1 with plane
42  // x=v0+u(v1-v0)
43  // (x-P0).n=0
44  // (v0+u(v1-v0)-P0).n=0
45  // u=(P0-v0).n/(v1-v0).n;
46  const double edgeNorm=(v1-v0).norm();
47  if(edgeNorm<FLT_EPSILON)
48  {
49  if(fabs((P0-0.5*(v0+v1)).dot(n))<FLT_EPSILON)
50  {
51  return std::make_tuple(INCIDENT,v0,v1);
52  }
53  else
54  {
55  return std::make_tuple(OFFSET,VectorDimD::Zero(),VectorDimD::Zero());
56  }
57  }
58  else
59  {
60  const double den=(v1-v0).dot(n);
61  const double num=(P0-v0).dot(n);
62  const double P0v0norm=(P0-v0).norm();
63 
64  const double numCheck= (P0v0norm<FLT_EPSILON)? 0.0 : num/P0v0norm;
65 
66  if (fabs(den/edgeNorm)>FLT_EPSILON)
67  {
68  // edge intersects plane
69  const double u=num/den;
70 
71  if(fabs(u)<FLT_EPSILON)
72  {
73  return std::make_tuple(INCIDENT,v0,v0);
74  }
75  else if (u>=FLT_EPSILON && u<=1.0-FLT_EPSILON)
76  {
77  const VectorDimD x((1.0-u)*v0 + u*v1);
78  return std::make_tuple(INCIDENT,x,x);
79  }
80  else if (fabs(1.0-u)<FLT_EPSILON)
81  {
82  return std::make_tuple(INCIDENT,v0,v0);
83  }
84  else
85  {// no roots
86  return std::make_tuple(OFFSET,VectorDimD::Zero(),VectorDimD::Zero());
87  }
88 
89  }
90  else
91  {
92  if (fabs(numCheck)>FLT_EPSILON)
93  {// edge is parallel to plane, no intersection
94  return std::make_tuple(PARALLEL,VectorDimD::Zero(),VectorDimD::Zero());
95 
96  }
97  else
98  {// edge is coplanar
99  return std::make_tuple(COINCIDENT,v0,v1);
100  }
101  }
102  }
103  }
