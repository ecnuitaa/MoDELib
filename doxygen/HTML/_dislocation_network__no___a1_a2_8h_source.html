184  void make_dt()
185  {
198  model::cout<<" Computing dt..."<<std::flush;
199  const auto t0=std::chrono::system_clock::now();
200 
201  // double vmax(0.0);
202  vmax=0.0;
203  int nVmean=0;
204  double vmean=0.0;
205  double dt_mean=0.0;
206 
207  for (const auto& nodeIter : this->nodes())
208  {
209  if(!nodeIter.second->isBoundaryNode() && !nodeIter.second->isConnectedToBoundaryNodes() && nodeIter.second->confiningPlanes().size()<3)
210  {
211  const double vNorm(nodeIter.second->get_V().norm());
212  vmean +=vNorm;
213  nVmean++;
214  if (vNorm>vmax)
215  {
216  vmax=vNorm;
217  }
218  }
219  }
220  vmean/=nVmean;
221 
222  //double shearWaveSpeedFraction(0.01);
223  //short unsigned int shearWaveExp=1;
224  if (vmax > Material<Isotropic>::cs*shearWaveSpeedFraction)
225  {
226  dt=dx/vmax;
227 
228  }
229  else
230  {
231  //dt=dx/std::pow(shared.material.cs*shearWaveSpeedFraction,shearWaveExp+1)*std::pow(vmax,shearWaveExp);
232  //dt=dx/(shared.material.cs*shearWaveSpeedFraction)*std::pow(vmax/(shared.material.cs*shearWaveSpeedFraction),1);
233  dt=dx/(Material<Isotropic>::cs*shearWaveSpeedFraction);
234  }
235 
236  if (vmean > Material<Isotropic>::cs*shearWaveSpeedFraction)
237  {
238  dt_mean=dx/vmean;
239  }
240  else
241  {
242  dt_mean=dx/(Material<Isotropic>::cs*shearWaveSpeedFraction);
243  }

245  model::cout<<std::setprecision(3)<<std::scientific<<" vmax="<<vmax;

247  model::cout<<std::setprecision(3)<<std::scientific<<" dt="<<dt;
248  model::cout<<std::setprecision(3)<<std::scientific<<" eta_dt="<<dt/dt_mean;
249  model::cout<<magentaColor<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t0)).count()<<" sec]"<<defaultColor<<std::endl;
