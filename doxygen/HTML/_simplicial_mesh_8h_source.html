387  /**********************************************************************/
388  std::pair<bool,const Simplex<dim,dim>*> isStrictlyInsideMesh(const Eigen::Matrix<double,dim,1>& P,
389  const Simplex<dim,dim>* const guess,
390  const double& tol) const
391  {
392  std::pair<bool,const Simplex<dim,dim>*> temp(searchWithGuess(P,guess));
393  if(temp.first)
394  {
395  const Eigen::Matrix<double,dim+1,1> bary(temp.second->pos2bary(P));
396  int kMin;
397  const double baryMin(bary.minCoeff(&kMin));
398  if (std::fabs(baryMin)<tol && guess->child(kMin).isBoundarySimplex()) // on a boundary face
399  {
400  temp.first=false;
401  }
402  }
403  return temp;
404  }
