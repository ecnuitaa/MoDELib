464  void createEdges(const EVLio<dim>& evl)
465  {
470  std::map<size_t,std::map<size_t,size_t>> loopMap;

473  loopMap[looplink.loopID].emplace(looplink.sourceID,looplink.sinkID);
474  }
475 
476 
477 
478  assert(loopMap.size()==evl.loops().size());
479 
480  size_t loopLumber=1;
481  for(const auto& loop : evl.loops())
482  {// for each loop in the EVLio<dim> object
483 
484  const auto loopFound=loopMap.find(loop.sID); // there must be an entry with key loopID in loopMap
485  assert(loopFound!=loopMap.end());
486  std::vector<size_t> nodeIDs;
487  nodeIDs.push_back(loopFound->second.begin()->first);
488  for(size_t k=0;k<loopFound->second.size();++k)
489  {
490  const auto nodeFound=loopFound->second.find(*nodeIDs.rbegin());
491  if(k<loopFound->second.size()-1)
492  {
493  nodeIDs.push_back(nodeFound->second);
494  }
495  else
496  {
497  assert(nodeFound->second==nodeIDs[0]);
498  }
499  }
