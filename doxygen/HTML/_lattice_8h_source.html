152  /**********************************************************************/
153  static Eigen::Matrix<long int,dim,1> rationalApproximation(VectorDimD nd)
154  {
155  Eigen::Array<long int,dim,1> nums=Eigen::Matrix<long int,dim,1>::Zero();
156 
157  if(nd.squaredNorm()>0.0)
158  {
159  const Eigen::Array<double,dim,1> nda(nd.array().abs()); // vector of close-to-integer numbers corresponding to lattice coordinates
160  size_t maxID=0;
161  const double maxVal(nda.maxCoeff(&maxID));
162  nd/=maxVal; // make each value of nd in [-1:1]
163 
164  nums=Eigen::Matrix<long int,dim,1>::Ones();
165  Eigen::Array<long int,dim,1> dens=Eigen::Matrix<long int,dim,1>::Ones();
166  long int denProd=1;
167 
168  for(int k=0;k<dim;++k)
169  {
170  BestRationalApproximation bra(nd(k),10000);
171 
172  nums(k)=bra.num;
173  dens(k)=bra.den;
174  denProd*=bra.den;
175  }
176 
177  for(int k=0;k<dim;++k)
178  {
179  nums(k)*=(denProd/dens(k));
180  }
181  }
182 
183 
184 
185  return nums.matrix();
186  }
