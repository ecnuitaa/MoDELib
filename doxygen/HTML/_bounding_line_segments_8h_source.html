214  static void emplaceUnique(LineSegmentContainerType& temp,const VectorDim& P1,const VectorDim& P2)
215  {
216 
217 // std::cout<<"emplaceUnique "<<P1.transpose()<<" ### "<<P2.transpose()<<std::endl;
218 
219  bool isUnique=true;
220 
221 
222  if((P1-P2).squaredNorm()>FLT_EPSILON)
223  {// P1-P2 is not a degenerate line
224  for(typename LineSegmentContainerType::iterator iter=temp.begin(); iter!=temp.end();)
225  {
226 
227  VectorDim& E1(iter->first);
228  VectorDim& E2(iter->second);
229  if((E1-E2).squaredNorm()>FLT_EPSILON)
230  {// existing pair is not a degenerate line
231  // isUnique*=((E1-P1).squaredNorm()>FLT_EPSILON && (E2-P2).squaredNorm()>FLT_EPSILON);
232  isUnique*=((E1-P1).squaredNorm()>FLT_EPSILON || (E2-P2).squaredNorm()>FLT_EPSILON);
233  isUnique*=((E1-P2).squaredNorm()>FLT_EPSILON || (E2-P1).squaredNorm()>FLT_EPSILON);
234 // std::cout<<" A existing "<<E1.transpose()<<" ### "<<E2.transpose()<<" ### "<<isUnique<<std::endl;
235  if(!isUnique)
236  {
237  break;
238  }
239  else
240  {
241  iter++;
242  }
243  }
244  else
245  {// existing pair is a degenerate line
246  const VectorDim x=0.5*(E1+E2);
247  if((P1-x).squaredNorm()>FLT_EPSILON && (P2-x).squaredNorm()>FLT_EPSILON)
248  {// distinct point
249  isUnique*=true;
250 // std::cout<<" B existing "<<E1.transpose()<<" ### "<<E2.transpose()<<" ### "<<isUnique<<std::endl;
251  iter++;
252  }
253  else
254  {// existing degenerate point is erased since P1-P2 will be appended later
255  isUnique*=true;
256 // std::cout<<" C existing "<<E1.transpose()<<" ### "<<E2.transpose()<<" ### "<<isUnique<<std::endl;
257  iter=temp.erase(iter);
258  }
259  }
260  }
261  }
262  else
263  {// P1-P2 is a degenerate line (a point)
264  const VectorDim x=0.5*(P1+P2);
265  for(const auto& pair : temp)
266  {
267  isUnique*=((pair.first-x).squaredNorm()>FLT_EPSILON && (pair.second-x).squaredNorm()>FLT_EPSILON);
268 // std::cout<<" D existing "<<pair.first.transpose()<<" ### "<<pair.second.transpose()<<" ### "<<isUnique<<std::endl;
269  if(!isUnique)
270  {
271  break;
272  }
273  }
274 
275  }
276 
277  if(isUnique)
278  {
279  temp.emplace_back(P1,P2);
280  }
281  else
282  {
283  // std::cout<<"BoundingLineSegments: not unique"<<std::endl;
284  }
285 
286  }
