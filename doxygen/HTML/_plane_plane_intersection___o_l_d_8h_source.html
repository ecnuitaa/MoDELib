48  LatticeVector<3> midPoint(const LatticePlane& plane1,
49  const LatticePlane& plane2,
50  const LatticeDirection<3>& dir)
51  {
52 
53  const LatticeVector<3>& p1=plane1.P;
54  const ReciprocalLatticeDirection<3>& n1=plane1.n;
55  const LatticeVector<3>& p2=plane2.P;
56  const ReciprocalLatticeDirection<3>& n2=plane2.n;
57 
58  // Find the Cartesian point P which minimizes (P-p1)^2+(P-p2)^2 under
59  // the constraints (P-P1)*n1=0 and (P-P2)*n2=0
60  Eigen::Matrix<double,5,5> M(Eigen::Matrix<double,5,5>::Zero());
61  M.block<3,3>(0,0)=2.0*Eigen::Matrix<double,3,3>::Identity();
62  M.block<3,1>(0,3)=n1.cartesian();
63  M.block<3,1>(0,4)=n2.cartesian();
64  M.block<1,3>(3,0)=n1.cartesian().transpose();
65  M.block<1,3>(4,0)=n2.cartesian().transpose();
66 
67  Eigen::Matrix<double,5,1> b(Eigen::Matrix<double,5,1>::Zero());
68  b.segment<3>(0)=p1.cartesian()+p2.cartesian();
69  b(3)=p1.cartesian().dot(n1.cartesian());
70  b(4)=p2.cartesian().dot(n2.cartesian());
71 
72  const Eigen::Matrix<double,3,1> C=M.lu().solve(b).segment<3>(0);
73 
74  assert(fabs((C-p1.cartesian()).dot(n1.cartesian()))<FLT_EPSILON);
75  assert(fabs((C-p2.cartesian()).dot(n2.cartesian()))<FLT_EPSILON);
76 
77  // Starting from C, move along dir and snap o lattice
78  LatticeVector<3> temp(p1);
79  const Eigen::Matrix<double,3,1> dc=dir.cartesian();
80  const int kMax=100;
81  for (int k=0;k<kMax;++k)
82  {
83  temp=plane1.snapToLattice(C+k*1.0/kMax*dc);
84  if(plane2.contains(temp))
85  {
86  break;
87  }
88  }
89 
90  //assert((temp-p1).dot(n1)==0);
91  //assert((temp-p2).dot(n2)==0);
92 
93 
94  return temp;
95 
96  }
