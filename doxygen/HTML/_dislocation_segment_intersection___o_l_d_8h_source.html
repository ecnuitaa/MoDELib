48 
49  bool areParallelNormals(normal1.cross(normal2).norm()<FLT_EPSILON);
50  bool areCoincidentPoints(std::fabs((pt1-pt2).dot(normal1))<FLT_EPSILON);
51 
52  int i;
53  if(areParallelNormals && !areCoincidentPoints)
54  {
55  /*parallel planes: no intersection*/
56  i= 0;
57  }
58  else if(areParallelNormals && areCoincidentPoints)
59  {
60  /*unique planes: planar intersection intersection*/
61  i= 1;
62  }
63  else
64  {
65  /*angle planes: angular intersection*/
66  i= 2;
67  }
68 
69  return i;
