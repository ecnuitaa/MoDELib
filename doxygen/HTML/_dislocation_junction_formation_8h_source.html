269  size_t junctionStep(const double& dx)
270  {
271 
272 #ifdef _OPENMP
273  const size_t nThreads = omp_get_max_threads();
274 #else
275  const size_t nThreads = 1;
276 #endif
277 
278  std::deque<IntersectionTypeContainerType> intersectionContainer;
279  intersectionContainer.resize(nThreads);
280  findIntersections(intersectionContainer,nThreads);
281 
282  const auto t0= std::chrono::system_clock::now();
283  model::cout<<" Forming Junctions: "<<std::flush;
284 
285  size_t nContracted=0;
286  for (const auto& intersectionByThreadContainer : intersectionContainer)
287  {
288  for (const auto& intersection : intersectionByThreadContainer)
289  {
290  const EdgeIDType& key1(std::get<0>(intersection));
291  const EdgeIDType& key2(std::get<1>(intersection));
292  const SegmentSegmentDistance<dim>& ssd(std::get<2>(intersection));
293  const double& t(ssd.t);
294  const double& u(ssd.u);
295 
296  const IsNetworkLinkType L1(DN.link(key1.first,key1.second));
297  const IsNetworkLinkType L2(DN.link(key2.first,key2.second));
298 
299  if(L1.first && L2.first) // Links exist
300  {
301 
302  auto Ni=L1.second->source;
303  if((ssd.x0-L1.second->source->get_P()).norm()>DislocationNetworkRemesh<DislocationNetworkType>::Lmin)
304  {
305  if((ssd.x0-L1.second->sink->get_P()).norm()>DislocationNetworkRemesh<DislocationNetworkType>::Lmin)
306  {
307  Ni=DN.expand(key1.first,key1.second,ssd.x0);
308  }
309  else
310  {
311  Ni=L1.second->sink;
312  }
313  }
314 
315  auto Nj=L2.second->source;
316  if((ssd.x1-L2.second->source->get_P()).norm()>DislocationNetworkRemesh<DislocationNetworkType>::Lmin)
317  {
318  if((ssd.x1-L2.second->sink->get_P()).norm()>DislocationNetworkRemesh<DislocationNetworkType>::Lmin)
319  {
320  Nj=DN.expand(key2.first,key2.second,ssd.x1);
321  }
322  else
323  {
324  Nj=L2.second->sink;
325  }
326  }
327 
328  VerboseJunctions(1,"forming junction "<<key1.first<<"->"<<key1.second<<", "
329  /* */ <<key2.first<<"->"<<key2.second<<", "
330  /* */ <<"dMin="<<ssd.dMin<<", "
331  /* */ <<"@ ("<<t<<","<<u<<"), "
332  /* */ <<"contracting "<<Ni->sID<<" "<<Nj->sID<<std::endl;);
333 
334  const bool success=DN.contract(Ni,Nj);
335  nContracted+=success;
336  if(!success)
337  {
338  // std::cout<<"IF CONTRACT DID NOT HAPPEN REMOVE NODES THAT WERE CREATED BY EXPANSION. STORE IDS IN CONTAINER AND THEN REMOVE"<<std::endl;
339  }
340  }
341  }
342  }
343  model::cout<<" ("<<nContracted<<" contracted)"<<magentaColor<<" ["<<(std::chrono::duration<double>(std::chrono::system_clock::now()-t0)).count()<<" sec]"<<defaultColor<<std::endl;
344  return nContracted;
345  }
